      ];
      const colors = barrierColors[index % barrierColors.length];
      
      const barrierGeometry = new THREE.CylinderGeometry(0.15, 0.15, 5.5, 16);
      const barrierMaterial = new THREE.MeshStandardMaterial({ 
        color: colors[0],
        roughness: 0.5
      });
      const barrier = new THREE.Mesh(barrierGeometry, barrierMaterial);
      barrier.rotation.z = Math.PI / 2;
      barrier.position.set(0, pos.height, 0);
      barrier.castShadow = true;
      obstacleGroup.add(barrier);
      
      for (let i = 0; i < 3; i++) {
        const stripe = new THREE.Mesh(
          new THREE.CylinderGeometry(0.16, 0.16, 0.8, 16),
          new THREE.MeshStandardMaterial({ color: colors[1], roughness: 0.5 })
        );
        stripe.rotation.z = Math.PI / 2;
        stripe.position.set(i * 1.5 - 1.5, pos.height, 0);
        obstacleGroup.add(stripe);
      }

      obstacleGroup.position.set(pos.x, 0, pos.z);
      scene.add(obstacleGroup);
      obstacles.push({ mesh: obstacleGroup, position: pos, cleared: false, height: pos.height });
    });

    gameStateRef.current.obstacles = obstacles;

    // Finish line with spectators
    const bannerGeometry = new THREE.PlaneGeometry(15, 3);
    const bannerMaterial = new THREE.MeshStandardMaterial({ 
      color: 0xFFFFFF,
      side: THREE.DoubleSide
    });
    const banner = new THREE.Mesh(bannerGeometry, bannerMaterial);
    banner.position.set(0, 4, 110);
    scene.add(banner);
    
    const finishText = new THREE.Mesh(
      new THREE.PlaneGeometry(10, 2),
      new THREE.MeshBasicMaterial({ color: 0xFF0000 })
    );
    finishText.position.set(0, 4, 109.9);
    scene.add(finishText);

    const createSpectator = (x, z) => {
      const spectatorGroup = new THREE.Group();
      
      const body = new THREE.Mesh(
        new THREE.CapsuleGeometry(0.3, 0.8, 8, 16),
        new THREE.MeshStandardMaterial({ 
          color: [0xFF6B6B, 0x4ECDC4, 0xFFE66D, 0x95E1D3, 0xFF8C00][Math.floor(Math.random() * 5)],
          roughness: 0.7
        })
      );
      body.position.y = 1.2;
      body.castShadow = true;
      spectatorGroup.add(body);
      
      const head = new THREE.Mesh(
        new THREE.SphereGeometry(0.25, 16, 16),
        new THREE.MeshStandardMaterial({ color: 0xffdbac, roughness: 0.8 })
      );
      head.position.y = 2;
      head.castShadow = true;
      spectatorGroup.add(head);
      
      [-0.35, 0.35].forEach(xOffset => {
        const arm = new THREE.Mesh(
          new THREE.CapsuleGeometry(0.1, 0.5, 6, 12),
          new THREE.MeshStandardMaterial({ 
            color: [0xFF6B6B, 0x4ECDC4, 0xFFE66D, 0x95E1D3, 0xFF8C00][Math.floor(Math.random() * 5)]
          })
        );
        arm.position.set(xOffset, 1.8, 0);
        arm.rotation.z = xOffset > 0 ? -0.8 : 0.8;
        arm.castShadow = true;
        spectatorGroup.add(arm);
      });
      
      spectatorGroup.position.set(x, 0, z);
      spectatorGroup.userData.originalY = spectatorGroup.position.y;
      spectatorGroup.userData.animOffset = Math.random() * Math.PI * 2;
      return spectatorGroup;
    };

    const spectators = [];
    for (let i = -20; i < 20; i += 3) {
      const leftSpectator = createSpectator(-15 + (Math.random() - 0.5) * 3, 105 + i);
      const rightSpectator = createSpectator(15 + (Math.random() - 0.5) * 3, 105 + i);
      scene.add(leftSpectator);
      scene.add(rightSpectator);
      spectators.push(leftSpectator, rightSpectator);
    }
    gameStateRef.current.spectators = spectators;

    camera.position.set(0, 10, -82);
    camera.lookAt(0, 2, -80);

    const keys = {};
    const handleKeyDown = (e) => { 
      if (e.key === 'PageUp') {
        keys['pageup'] = true;
      }
    };
    const handleKeyUp = (e) => { 
      if (e.key === 'PageUp') {
        keys['pageup'] = false;
      }
    };
    window.addEventListener('keydown', handleKeyDown);
    window.addEventListener('keyup', handleKeyUp);

    gameStateRef.current.startTime = Date.now();
    let animationId;
    let faultCount = 0;

    const animate = () => {
      if (gamePaused) {
        animationId = requestAnimationFrame(animate);
        return;
      }

      const horse = gameStateRef.current.horse;
      const velocity = gameStateRef.current.velocity;
      const legs = gameStateRef.current.legs;
      const tail = gameStateRef.current.tail;

      // FAST AUTO-RUN
      velocity.z = selectedRider.speed;

      // REALISTIC GALLOP ANIMATION
      gameStateRef.current.gallopPhase += 0.35;
      const gallopTime = gameStateRef.current.gallopPhase;
      
      if (legs && !gameStateRef.current.isJumping) {
        // Realistic gallop - legs move in diagonal pairs
        legs.forEach((leg, index) => {
          const phase = leg.isFront ? gallopTime : gallopTime + Math.PI;
          const isLeft = index % 2 === 0;
          const sidedness = isLeft ? 0 : Math.PI;
          
          // Forward/backward swing
          const swingAngle = Math.sin(phase + sidedness) * (leg.isFront ? 1 : 0.7);
          leg.group.rotation.x = swingAngle;
          
          // Knee bend during gallop
          leg.knee.position.y = -1.1 + Math.abs(Math.sin(phase + sidedness)) * 0.2;
          leg.lowerLeg.rotation.x = Math.max(0, Math.sin(phase + sidedness)) * 0.8;
        });
        
        // Horse body bobbing - more pronounced
        horse.position.y = 0.5 + Math.abs(Math.sin(gallopTime * 0.5)) * 0.35;
        horse.rotation.z = Math.sin(gallopTime * 0.5) * 0.08;
        
        // Head bobbing
        head.rotation.x = 0.2 + Math.sin(gallopTime * 0.5) * 0.15;
      }

      // Tail swaying dynamically
      if (tail) {
        tail.rotation.x = Math.sin(gallopTime * 0.4) * 0.4;
        tail.rotation.y = Math.sin(gallopTime * 0.3) * 0.2;
      }

      // Jump
      if (keys['pageup'] && !gameStateRef.current.isJumping && horse.position.y <= 0.85) {
        gameStateRef.current.isJumping = true;
        velocity.y = selectedRider.jumpPower;
        
        // Extend legs during jump
        if (legs) {
          legs.forEach(leg => {
            leg.group.rotation.x = -0.3;
          });
        }
      }

      horse.position.z += velocity.z;

      // Gravity
      if (gameStateRef.current.isJumping) {
        velocity.y -= 0.015;
        horse.position.y += velocity.y;
        
        if (horse.position.y <= 0.5) {
          horse.position.y = 0.5;
          gameStateRef.current.isJumping = false;
          velocity.y = 0;
        }
      }

      // Animate spectators
      const time = Date.now() * 0.003;
      spectators.forEach((spectator) => {
        const offset = spectator.userData.animOffset;
        spectator.position.y = Math.sin(time + offset) * 0.15;
        spectator.rotation.z = Math.sin(time * 2 + offset) * 0.1;
        
        spectator.children.forEach((child, index) => {
          if (index >= 2) {
            child.rotation.z = child.position.x > 0 
              ? -0.8 + Math.sin(time * 3 + offset) * 0.4
              : 0.8 - Math.sin(time * 3 + offset) * 0.4;
          }
        });
      });

      // Obstacle collision
      obstacles.forEach((obstacle, index) => {
        if (obstacle.cleared) return;

        const distance = Math.sqrt(
          Math.pow(horse.position.x - obstacle.position.x, 2) +
          Math.pow(horse.position.z - obstacle.position.z, 2)
        );

        if (distance < 3) {
          if (horse.position.y < obstacle.height - 0.3) {
            faultCount++;
            obstacle.mesh.rotation.x = 0.3;
            setTimeout(() => { obstacle.mesh.rotation.x = 0; }, 500);
          } else if (horse.position.z > obstacle.position.z + 2) {
            obstacle.cleared = true;
            gameStateRef.current.clearedObstacles.add(index);
          }
        }
      });

      // Dynamic camera follow
      const targetX = horse.position.x;
      const targetZ = horse.position.z - 15;
      const targetY = horse.position.y + 14;
      
      camera.position.x += (targetX - camera.position.x) * 0.12;
      camera.position.z += (targetZ - camera.position.z) * 0.12;
      camera.position.y += (targetY - camera.position.y) * 0.12;
      camera.lookAt(horse.position.x, horse.position.y + 2, horse.position.z + 10);

      const elapsed = (Date.now() - gameStateRef.current.startTime) / 1000;
      const points = Math.max(0, 1000 - faultCount * 100 - Math.floor(elapsed * 2));
      setScore({
        time: elapsed.toFixed(1),
        faults: faultCount,
        points: points
      });

      if (horse.position.z > 108) {
        setFinished(true);
        setGamePaused(true);
      }

      renderer.render(scene, camera);
      animationId = requestAnimationFrame(animate);
    };

    animate();

    const handleResize = () => {
      if (!containerRef.current) return;
      camera.aspect = containerRef.current.clientWidth / containerRef.current.clientHeight;
      camera.updateProjectionMatrix();
      renderer.setSize(containerRef.current.clientWidth, containerRef.current.clientHeight);
    };
    window.addEventListener('resize', handleResize);

    return () => {
      cancelAnimationFrame(animationId);
      window.removeEventListener('keydown', handleKeyDown);
      window.removeEventListener('keyup', handleKeyUp);
      window.removeEventListener('resize', handleResize);
      if (containerRef.current && renderer.domElement) {
        containerRef.current.removeChild(renderer.domElement);
      }
      renderer.dispose();
    };
  }, [selectedRider, raceStarted, gamePaused, horseCareComplete]);

  const handleStartGame = () => {
    if (!selectedRider) return;
    setGameStarted(true);
    setCountdown(5);
    setScore({ time: 0, faults: 0, points: 1000 });
    setFinished(false);
  };

  const handleRestart = () => {
    setGameStarted(false);
    setRaceStarted(false);
    setGamePaused(false);
    setFinished(false);
    setHorseCareComplete(false);
    setCountdown(5);
    setScore({ time: 0, faults: 0, points: 0 });
  };

  const handleHorseCareComplete = () => {
    setHorseCareComplete(true);
    setTimeout(() => handleStartGame(), 500);
  };

  if (!selectedRider) {
    return (
      <div className="min-h-screen bg-gradient-to-br from-green-50 via-blue-50 to-green-100 p-4 md:p-8">
        <div className="max-w-6xl mx-auto">
          <div className="text-center mb-12">
            <h1 className="text-4xl md:text-5xl font-bold text-gray-900 mb-4">
              üèá Show Jumping Champion
            </h1>
            <p className="text-lg md:text-xl text-gray-600">
              W√§hle deinen Reiter und springe zum Sieg!
            </p>
          </div>
          <CharacterSelector riders={RIDERS} onSelect={setSelectedRider} />
        </div>
      </div>
    );
  }

  if (!horseCareComplete) {
    return <HorseCare rider={selectedRider} onComplete={handleHorseCareComplete} onBack={() => setSelectedRider(null)} />;
  }

  if (gameStarted && countdown > 0) {
    return (
      <div className="relative w-full h-screen bg-gradient-to-br from-green-900 via-green-700 to-green-900 flex items-center justify-center overflow-hidden">
        <div className="absolute inset-0 bg-black/30" />
        <div className="relative z-10 text-center">
          <h2 className="text-white text-4xl md:text-6xl font-bold mb-8">
            Mach dich bereit!
          </h2>
          <div className="text-white text-9xl md:text-[200px] font-bold animate-pulse">
            {countdown}
          </div>
          <p className="text-white text-2xl md:text-3xl mt-8">
            Das Rennen beginnt gleich...
          </p>
        </div>
      </div>
    );
  }

  return (
    <div className="relative w-full h-screen bg-black overflow-hidden">
      <div ref={containerRef} className="w-full h-full" />
      
      <ScoreBoard score={score} finished={finished} />

      <div className="absolute top-4 right-4 flex gap-2">
        <Button
          onClick={() => setGamePaused(!gamePaused)}
          variant="secondary"
          size="icon"
          className="bg-white/90 backdrop-blur"
        >
          {gamePaused ? <Play className="w-5 h-5" /> : <Pause className="w-5 h-5" />}
        </Button>
        <Button
          onClick={handleRestart}
          variant="secondary"
          size="icon"
          className="bg-white/90 backdrop-blur"
        >
          <RotateCcw className="w-5 h-5" />
        </Button>
      </div>

      <div className="absolute bottom-8 left-1/2 transform -translate-x-1/2">
        <Card className="bg-white/95 backdrop-blur border-2 border-green-500 shadow-2xl">
          <CardContent className="p-4">
            <p className="text-xl font-bold text-center text-green-600">
              üéÆ Dr√ºcke PgUp (Bild ‚Üë) zum Springen!
            </p>
          </CardContent>
        </Card>
      </div>

      {finished && (
        <div className="absolute inset-0 bg-black/70 backdrop-blur-sm flex items-center justify-center p-4">
          <Card className="w-full max-w-md">
            <CardContent className="p-8 text-center space-y-6">
              <div className="text-6xl mb-4">üéâ</div>
              <div className="w-20 h-20 mx-auto bg-yellow-100 rounded-full flex items-center justify-center">
                <Trophy className="w-10 h-10 text-yellow-600" />
              </div>
              <h2 className="text-3xl font-bold">Gl√ºckwunsch!</h2>
              <p className="text-xl text-gray-600">Die Zuschauer jubeln! üéä</p>
              <div className="space-y-3">
                <div className="bg-gray-50 p-4 rounded-lg">
                  <p className="text-sm text-gray-600">Zeit</p>
                  <p className="text-2xl font-bold">{score.time}s</p>
                </div>
                <div className="bg-gray-50 p-4 rounded-lg">
                  <p className="text-sm text-gray-600">Fehler</p>
                  <p className="text-2xl font-bold">{score.faults}</p>
                </div>
                <div className="bg-green-50 p-4 rounded-lg">
                  <p className="text-sm text-gray-600">Gesamtpunkte</p>
                  <p className="text-3xl font-bold text-green-600">{score.points}</p>
                </div>
              </div>
              <div className="space-y-2">
                <Button onClick={handleRestart} className="w-full" size="lg">
                  <RotateCcw className="w-5 h-5 mr-2" />
                  Nochmal spielen
                </Button>
                <Button onClick={() => { setSelectedRider(null); setHorseCareComplete(false); }} variant="outline" className="w-full">
                  Reiter wechseln
                </Button>
              </div>
            </CardContent>
          </Card>
        </div>
      )}
    </div>
  );
}
